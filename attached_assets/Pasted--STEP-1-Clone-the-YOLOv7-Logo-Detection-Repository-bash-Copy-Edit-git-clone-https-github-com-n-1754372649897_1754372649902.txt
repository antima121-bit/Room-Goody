🧠 STEP 3: Create app.py (Flask REST API)

Copy
Edit
from flask import Flask, request, jsonify
import os
from detect import run  # Import the detect function from YOLOv7
import uuid

app = Flask(__name__)

@app.route('/detect-logo', methods=['POST'])
def detect_logo():
    if 'image' not in request.files:
        return jsonify({'error': 'No image uploaded'}), 400

    image_file = request.files['image']
    image_path = f"uploads/{uuid.uuid4().hex}.jpg"
    os.makedirs("uploads", exist_ok=True)
    image_file.save(image_path)

    # Run YOLOv7 detection
    results = run(
        weights='yolov7.pt',
        source=image_path,
        conf_thres=0.25,
        save_txt=False,
        save_conf=True,
        nosave=False
    )

    # Read output from runs/detect/exp/labels (if needed)
    # Or return static response for now
    return jsonify({
        "status": "success",
        "detected": True,
        "image": image_path
        # Optionally, parse prediction results and return label, box, confidence
    })

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=5000)
