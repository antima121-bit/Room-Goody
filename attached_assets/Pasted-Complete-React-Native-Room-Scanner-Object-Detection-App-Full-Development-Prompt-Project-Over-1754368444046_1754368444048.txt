-- Detected objects table

CREATE TABLE detected_objects (

ml_metadata TEXT NOT NULL, -- JSON object

gps_location TEXT NOT NULL, -- JSON object

colors TEXT NOT NULL, -- JSON array

sync_status TEXT DEFAULT 'pending'

);



-- Search cache table

CREATE TABLE search_cache (

query_hash TEXT PRIMARY KEY,

results TEXT NOT NULL, -- JSON array

);

Cloud Storage (Supabase)

-- Users table (extends auth.users)

CREATE TABLE user_profiles (

id UUID REFERENCES auth.users(id) PRIMARY KEY,

first_name TEXT,

last_name TEXT,

avatar_url TEXT,

settings JSONB DEFAULT '{}',

created_at TIMESTAMP DEFAULT NOW(),

updated_at TIMESTAMP DEFAULT NOW()

);



-- Rooms table

CREATE TABLE rooms (

id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

name TEXT NOT NULL,

user_id UUID REFERENCES auth.users(id),

layout_image_uri TEXT,

plan_data JSONB,

furniture JSONB DEFAULT '[]',

gps_location JSONB,

created_at TIMESTAMP DEFAULT NOW(),

updated_at TIMESTAMP DEFAULT NOW(),


-- Enable Row Level Security

CONSTRAINT rooms_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)

);



-- Detected objects table

CREATE TABLE detected_objects (

id UUID PRIMARY KEY DEFAULT gen_random_uuid(),

user_id UUID REFERENCES auth.users(id),

image_uri TEXT NOT NULL,

objects JSONB NOT NULL,

ml_metadata JSONB NOT NULL,

gps_location JSONB NOT NULL,

colors JSONB NOT NULL,

shapes JSONB NOT NULL,

created_at TIMESTAMP DEFAULT NOW(),


CONSTRAINT detected_objects_user_id_fkey FOREIGN KEY (user_id) REFERENCES auth.users(id)

);



-- Enable Row Level Security

ALTER TABLE rooms ENABLE ROW LEVEL SECURITY;

ALTER TABLE detected_objects ENABLE ROW LEVEL SECURITY;



-- RLS Policies

CREATE POLICY "Users can only access their own rooms" ON rooms

FOR ALL USING (auth.uid() = user_id);



CREATE POLICY "Users can only access their own objects" ON detected_objects

FOR ALL USING (auth.uid() = user_id);

Synchronization Strategy

Offline-first approach: All operations work offline

Background sync: Automatic sync when connected

Conflict resolution: Last-write-wins with user notification

Partial sync: Sync only changed data

Retry mechanism: Exponential backoff for failed syncs
